# 解法一：深度优先搜索遍历

```cpp
vector<vector<bool>> visit(55, vector<bool>(55, false));
int dir[4][2] = { 0, 1, 1, 0, 0, -1, -1, 0 };
int m, n, cnt;  // m 为行数，b 为列数，cnt 统计每块岛屿的面积

void DFS(const vector<vector<int>>& grid, int x, int y)
{
    ++cnt;  // 每遍历一个节点计数+1
    for (int i = 0; i < 4; ++i)
    {
        int tmpx = x + dir[i][0];
        int tmpy = y + dir[i][1];
        if (tmpx < 0 || tmpx >= m || tmpy < 0 || tmpy >= n) continue;
        if (visit[tmpx][tmpy] == false && grid[tmpx][tmpy] == 1)
        {
            visit[tmpx][tmpy] = true;
            DFS(grid, tmpx, tmpy);
        }
    }
}

int maxAreaOfIsland(vector<vector<int>>& grid)
{
    m = grid.size();
    n = grid[0].size();
    int res = 0;
    for (int i = 0; i < m; ++i)
    {
        for (int j = 0; j < n; ++j)
        {
            if (visit[i][j] == false && grid[i][j] == 1)
            {
                cnt = 0;    // 计数值置0
                visit[i][j] = true;
                DFS(grid, i, j);
                res = max(res, cnt);
            }
        }
    }

    return res;
}
```

# 解法二：广度优先搜索遍历

```cpp
vector<vector<bool>> visit(55, vector<bool>(55, false));
int dir[4][2] = { 0, 1, 1, 0, 0, -1, -1, 0 };
int m, n, cnt;  // m 为行数，b 为列数，cnt 统计每块岛屿的面积

void BFS(const vector<vector<int>>& grid, int x, int y)
{
    queue<pair<int, int>> qu;
    // qu.push({x, y}); // 这两行代码都在队列中插入一个 pair
    qu.push(make_pair(x, y));
    visit[x][y] = true;
    while (qu.empty() == false)
    {
        pair<int, int> tmp = qu.front();
        qu.pop();
        for (int i = 0; i < 4; ++i)
        {
            int tmpx = tmp.first + dir[i][0];
            int tmpy = tmp.second + dir[i][1];
            if (tmpx < 0 || tmpx >= m || tmpy < 0 || tmpy >= n) continue;
            if (visit[tmpx][tmpy] == false && grid[tmpx][tmpy] == 1)
            {
                ++cnt;
                qu.push(make_pair(tmpx, tmpy));
                visit[tmpx][tmpy] = true;
            }
        }
    }
}

int maxAreaOfIsland(vector<vector<int>>& grid)
{
    m = grid.size();
    n = grid[0].size();
    int res = 0;
    for (int i = 0; i < m; ++i)
    {
        for (int j = 0; j < n; ++j)
        {
            if (visit[i][j] == false && grid[i][j] == 1)
            {
                cnt = 1;
                BFS(grid, i, j);
                res = max(res, cnt);
            }
        }
    }

    return res;
}
```
