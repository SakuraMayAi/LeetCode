# 解法：回溯

```cpp
vector<vector<int>> res;
vector<int> path;
int used[21] = { 0 };

void Backtracking(vector<int>& nums, int start)
{
    if (path.size() == nums.size())
    {
        res.push_back(path);
        return;
    }
    for (int i = 0; i < nums.size(); ++i)
    {
        if (!used[nums[i] + 10])
        {
            used[nums[i] + 10] = 1;
            path.push_back(nums[i]);
            Backtracking(nums, i + 1);
            path.pop_back();
            used[nums[i] + 10] = 0;
        }
    }
}

vector<vector<int>> permute(vector<int>& nums)
{
    Backtracking(nums, 0);
    return res;
}
```
