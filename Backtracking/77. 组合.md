# 解法：回溯

```cpp
vector<vector<int>> res;
vector<int> path;

void BackTracking(int n, int k, int start)
{
    if (path.size() == k)
    {
        res.push_back(path);
        return;
    }
    for (int i = start; i <= n - (k - path.size()) + 1; ++i)
    {
        path.push_back(i);
        BackTracking(n, k, i + 1);
        path.pop_back();
    }
}

vector<vector<int>> combine(int n, int k)
{
    BackTracking(n, k, 1);
    return res;
}
```
